<script>
    let currentSlide = 1;
    const totalSlides = 8;
    let isInQuiz = false;
    let isSpeaking = false;

    const speakerNotes = { /* your full notes here - same as before */ };

    let audioPlayed = {};

    function updateUI() {
        document.getElementById('prevBtn').style.visibility = (currentSlide === 1 && !isInQuiz) ? 'hidden' : 'visible';
        document.getElementById('slideImg').src = `Slide${currentSlide}.JPG`;
        updateAudioButton();
    }

    function updateAudioButton() {
        const btn = document.getElementById('audioBtn');
        btn.textContent = audioPlayed[currentSlide] ? "🔊 Replay Audio" : "🔊 Play Audio";
    }

    function toggleAudio() {
        const btn = document.getElementById('audioBtn');
        if (isSpeaking) {
            window.speechSynthesis.cancel();
            isSpeaking = false;
            updateAudioButton();
        } else {
            speakNotes();
        }
    }

    function speakNotes() {
        const btn = document.getElementById('audioBtn');
        const text = speakerNotes[currentSlide];
        if (!text) return;

        window.speechSynthesis.cancel();
        btn.textContent = "🔊 Speaking...";
        isSpeaking = true;

        const sentences = text.match(/[^.!?]+[.!?]+(?:\s|$)/g) || [text];
        let index = 0;

        const speakNext = () => {
            if (index >= sentences.length) {
                audioPlayed[currentSlide] = true;
                btn.textContent = "🔊 Replay Audio";
                isSpeaking = false;
                return;
            }
            const u = new SpeechSynthesisUtterance(sentences[index].trim());
            u.rate = 0.93;
            u.onend = () => { index++; speakNext(); };
            u.onerror = () => { isSpeaking = false; updateAudioButton(); };
            window.speechSynthesis.speak(u);
        };
        speakNext();
    }

    function goBack() {
        if (isInQuiz) {
            exitQuiz();
        } else {
            changeSlide(-1);
        }
    }

    function nextAction() {
        if (currentSlide === totalSlides) {
            showQuiz();
        } else {
            changeSlide(1);
        }
    }

    function changeSlide(dir) {
        currentSlide = Math.max(1, Math.min(totalSlides, currentSlide + dir));
        window.speechSynthesis.cancel();
        updateUI();
    }

    function showQuiz() {
        isInQuiz = true;
        document.getElementById('slideImg').style.display = 'none';
        document.getElementById('quizStage').style.display = 'block';
        document.getElementById('mainControls').style.display = 'none'; // Hide all bottom buttons
        buildQuiz();
    }

    function exitQuiz() {
        isInQuiz = false;
        document.getElementById('quizStage').style.display = 'none';
        document.getElementById('slideImg').style.display = 'block';
        document.getElementById('mainControls').style.display = 'flex';
        updateUI();
    }

    // Quiz
    const quizQuestions = [
        { q: "1. Is 'jury nullification' officially recognised as a formal legal right?", correct: false },
        { q: "2. Do jury deliberations take place in secret?", correct: true },
        { q: "3. Can jurors be punished for their verdict?", correct: false },
        { q: "4. Is this debate a recent development?", correct: false },
        { q: "5. Does the jury act as a safeguard against unjust laws?", correct: true },
        { q: "6. Has the law given a single clear answer on jury conscience?", correct: false }
    ];

    let quizAnswers = [];

    function buildQuiz() {
        quizAnswers = new Array(quizQuestions.length).fill(null);
        const container = document.getElementById('quizContent');
        container.innerHTML = '';

        quizQuestions.forEach((q, i) => {
            const block = document.createElement('div');
            block.className = 'question-block';
            block.innerHTML = `<div style="font-weight:600; margin-bottom:12px;">${q.q}</div>`;

            const yesBtn = document.createElement('button');
            yesBtn.className = 'option-btn';
            yesBtn.textContent = 'Yes ✓';
            yesBtn.onclick = () => selectAnswer(i, true, yesBtn, block);

            const noBtn = document.createElement('button');
            noBtn.className = 'option-btn';
            noBtn.textContent = 'No ✕';
            noBtn.onclick = () => selectAnswer(i, false, noBtn, block);

            block.appendChild(yesBtn);
            block.appendChild(noBtn);
            container.appendChild(block);
        });

        const submitBtn = document.createElement('button');
        submitBtn.className = 'btn btn-primary';
        submitBtn.style.margin = '30px auto';
        submitBtn.style.display = 'block';
        submitBtn.textContent = 'Submit Quiz';
        submitBtn.onclick = submitQuiz;
        container.appendChild(submitBtn);
    }

    function selectAnswer(index, value, clickedBtn, block) {
        quizAnswers[index] = value;
        const buttons = block.querySelectorAll('.option-btn');
        buttons.forEach(b => b.classList.remove('selected'));
        clickedBtn.classList.add('selected');
    }

    function submitQuiz() {
        if (quizAnswers.some(a => a === null)) {
            alert("⚠️ Please answer ALL questions before submitting.");
            return;
        }

        let score = 0;
        const container = document.getElementById('quizContent');

        quizQuestions.forEach((q, i) => {
            if (quizAnswers[i] === q.correct) score++;

            const block = container.children[i];
            const buttons = block.querySelectorAll('.option-btn');
            buttons.forEach(btn => {
                const isYes = btn.textContent.includes('Yes');
                if (isYes === q.correct) btn.classList.add('correct');
                else if (isYes === quizAnswers[i]) btn.classList.add('wrong');
            });
        });

        const result = document.createElement('div');
        result.style.textAlign = 'center';
        result.style.margin = '30px 0';
        result.style.fontSize = '1.5rem';
        result.innerHTML = score === 6 ? '🎉 Perfect Score! Excellent work!' : `You scored ${score}/6`;
        container.appendChild(result);
    }

    function restartPresentation() {
        currentSlide = 1;
        isInQuiz = false;
        document.getElementById('completionScreen').style.display = 'none';
        document.getElementById('mainControls').style.display = 'flex';
        updateUI();
    }

    function restartQuiz() {
        isInQuiz = true;
        document.getElementById('completionScreen').style.display = 'none';
        showQuiz();
    }

    function finishPresentation() {
        window.close();
    }

    window.onload = updateUI;
</script>
