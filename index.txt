<script>
    let currentSlide = 1;
    const totalSlides = 8;

    function updateUI() {
        document.getElementById('slideNum').textContent = `Slide ${currentSlide} / ${totalSlides}`;
        const img = document.getElementById('slideImg');
        img.src = `Slide${currentSlide}.JPG`;   // Try .JPG first
    }

    function changeSlide(dir) {
        currentSlide = Math.max(1, Math.min(totalSlides, currentSlide + dir));
        updateUI();
    }

    window.onload = updateUI;
</script>
